    Sample StageStepper shape (place inside TopBar.tsx):

    ```tsx
    interface StageStepperProps {
      summary: PrSummary | null | undefined;
      selfReview: SelfReview | null | undefined;
      activeCategory: ChecklistCategory | null;
      onSummaryStep: () => void;
      onSelfReviewStep: () => void;
      onCategoryClick: (cat: ChecklistCategory | null) => void;
    }

    function StageStepper({ summary, selfReview, activeCategory, onSummaryStep, onSelfReviewStep, onCategoryClick }: StageStepperProps) {
      // Derive step states per UI-SPEC matrix
      const summaryState = summary ? 'done' : (!selfReview ? 'active' : 'done');
      const selfReviewState = selfReview ? 'done' : (summary ? 'active' : 'default');

      return (
        <div className="stages" role="list" aria-label="Review stages">
          <div className={`stage ${summaryState}`} role="listitem" aria-current={summaryState === 'active' ? 'step' : undefined}>
            <div className="number">1</div>
            <div className="label">Summary</div>
            <div className="sub">
              {summary
                ? `${summary.intent} · ${Math.round(summary.intentConfidence * 100)}% confident`
                : 'Not generated'}
            </div>
            {summary ? <button onClick={onSummaryStep} aria-label="Open summary drawer">…</button> : null}
          </div>
          {/* ... step 2, 3, 4 similarly ... */}
          {selfReview && (
            <div className="stages-coverage-strip" role="group" aria-label="Category coverage">
              {(['correctness', 'security', 'tests', 'performance', 'style'] as const).map((cat) => {
                const count = selfReview.findings.filter((f) => f.category === cat).length;
                const coverageStatus = selfReview.coverage[cat] ?? 'pass';
                const isActive = activeCategory === cat;
                return (
                  <button
                    key={cat}
                    className={`coverage-chip coverage-chip--${coverageStatus} ${isActive ? 'active' : ''}`}
                    onClick={() => onCategoryClick(isActive ? null : cat)}
                    aria-label={`${cat}: ${coverageStatus}`}
                  >
                    {cat.charAt(0).toUpperCase() + cat.slice(1)} ({count})
                  </button>
                );
              })}
            </div>
          )}
        </div>
      );
    }
    ```
