### `server/src/session/request-queue.ts` (utility, event-driven)

**Analog:** None — first event-driven Promise-resolver module in the codebase.

**Pattern from RESEARCH.md** (Pattern 2, verified):

No imports needed beyond Node built-ins. The module exports two classes and a shared type.

**Core pattern** (from RESEARCH.md §Area 2):

```typescript
export interface QueuedRequest {
  type: 'chat' | 'inline_comment' | 'run_self_review' | 'regenerate_summary' | 'regenerate_walkthrough';
  payload?: Record<string, unknown>;
}

export class SessionRequestQueue {
  private queue: QueuedRequest[] = [];
  private resolver: ((req: QueuedRequest | null) => void) | null = null;
  private timeout: ReturnType<typeof setTimeout> | null = null;

  enqueue(req: QueuedRequest): void {
    if (this.resolver !== null) {
      const resolve = this.resolver;
      this.resolver = null;          // clear BEFORE calling resolve (prevents double-resolve)
      if (this.timeout !== null) {
        clearTimeout(this.timeout);
        this.timeout = null;
      }
      resolve(req);
    } else {
      this.queue.push(req);
    }
  }

  async waitForRequest(ms: number): Promise<QueuedRequest | null> {
    if (this.queue.length > 0) return this.queue.shift()!;
    if (this.resolver !== null) throw new Error('already waiting');
    return new Promise((resolve) => {
      this.resolver = resolve;
      this.timeout = setTimeout(() => {
        this.resolver = null;
        this.timeout = null;
        resolve(null);  // null → tool returns { type: 'no_request' }
      }, ms);
    });
  }

  get pendingCount(): number { return this.queue.length; }
}

export class RequestQueueManager {
  private queues = new Map<string, SessionRequestQueue>();

  getQueue(prKey: string): SessionRequestQueue {
    if (!this.queues.has(prKey)) {
      this.queues.set(prKey, new SessionRequestQueue());
    }
    return this.queues.get(prKey)!;
  }
}
```

**Key invariants:**
- `enqueue` must clear `this.resolver = null` BEFORE calling `resolve(req)` — prevents double-resolve race
- `waitForRequest` throws if a resolver is already pending (concurrent call guard)
- On timeout: `resolve(null)` → tool handler maps to `{ type: 'no_request' }`
- Module does NOT call `applyEvent` — HTTP route and MCP handlers own that

export async function postUserRequest(
  prKey: string,
  req: {
    type: 'chat' | 'inline_comment' | 'run_self_review' | 'regenerate_summary' | 'regenerate_walkthrough';
    payload?: Record<string, unknown>;
  }

  if (!reviewToken) {
    throw new Error('postUserRequest: review token not set — call setReviewToken first');
  }
  const res = await fetch('/api/user-request', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-Review-Token': reviewToken,
    },
    body: JSON.stringify({ prKey, ...req }),
    credentials: 'same-origin',
  });
  if (!res.ok) {
    const text = await res.text().catch(() => '');
    throw new Error(`postUserRequest failed: HTTP ${res.status} ${text}`);
  }
  return res.json();
}
