      import type { SnapshotMessage, UpdateMessage } from '@shared/types';

      export function openEventStream(
        sessionKey: string,
        onSnapshot: (msg: SnapshotMessage) => void,
        onUpdate: (msg: UpdateMessage) => void,
        onError: () => void
      ): () => void {
        const es = new EventSource(`/api/events?session=${encodeURIComponent(sessionKey)}`, {
          withCredentials: true,
        });

        const snapshotHandler = (ev: MessageEvent) => {
          try { onSnapshot(JSON.parse(ev.data) as SnapshotMessage); }
          catch { onError(); }
        };
        const updateHandler = (ev: MessageEvent) => {
          try { onUpdate(JSON.parse(ev.data) as UpdateMessage); }
          catch { onError(); }
        };

        es.addEventListener('snapshot', snapshotHandler as EventListener);
        es.addEventListener('update', updateHandler as EventListener);
        es.onerror = onError;

        return () => es.close();
      }
      ```

           <div className="bg-white dark:bg-zinc-900 p-6 rounded-lg max-w-md shadow-xl">
             <p className="text-sm">Resume state incomplete — please re-run /review.</p>
           </div>
         </div>
       );
     }

     const handle = async (choice: ChooseResumeChoice) => {
       if (pending) return;
       setPending(choice);
       setError(null);
       try {
         await chooseResume({
           prKey: state.sessionKey,
           choice,
           source: state.source!,
         });
         // Server pushes snapshot/update via SSE; store reducer clears state.staleDiff;
         // modal unmounts naturally. If SSE never arrives, `pending` stays set and
         // "Refreshing diff…" is shown indefinitely — user can close the browser.
       } catch (err) {
         setError(err instanceof Error ? err.message : 'Resume failed');
         setPending(null);
       }
     };

     return (
       <div
         className="fixed inset-0 bg-black/70 flex items-center justify-center z-50"
         role="dialog"
         aria-modal="true"
         aria-labelledby="stale-diff-title"
       >
         <div className="bg-white dark:bg-zinc-900 p-6 rounded-lg max-w-md w-full shadow-xl">
           <h2 id="stale-diff-title" className="text-lg font-semibold mb-2">PR updated</h2>
           <p className="text-sm text-zinc-600 dark:text-zinc-400 mb-1">
             This PR's head SHA changed since your last session.
           </p>
           <p className="text-xs font-mono text-zinc-500 mb-4">
             Stored: {state.staleDiff.storedSha.slice(0, 8)} → Current: {state.staleDiff.currentSha.slice(0, 8)}
           </p>

           {pending ? (
             <div className="py-4 text-center text-sm text-zinc-600 dark:text-zinc-400">
               Refreshing diff…
             </div>
           ) : (
             <div className="flex flex-col gap-2">
               <button
                 className="px-4 py-2 bg-blue-600 hover:bg-blue-700 text-white rounded disabled:opacity-50"
                 onClick={() => handle('adopt')}
                 disabled={pending !== null}
               >
                 Refresh to current PR
               </button>
               <button
                 className="px-4 py-2 bg-red-600 hover:bg-red-700 text-white rounded disabled:opacity-50"
                 onClick={() => handle('reset')}
                 disabled={pending !== null}
               >
                 Discard session
               </button>
               <button
                 className="px-4 py-2 bg-zinc-200 dark:bg-zinc-700 hover:bg-zinc-300 dark:hover:bg-zinc-600 rounded disabled:opacity-50"
                 onClick={() => handle('viewBoth')}
                 disabled={pending !== null}
               >
                 View both
               </button>
             </div>
           )}

           {error && (
             <p className="mt-3 text-xs text-red-600">{error}</p>
           )}
         </div>
       </div>
     );
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → server POST `/api/session/choose-resume` | New client-initiated mutation path (Plan 03 defined the server side; this plan adds the client). Server-side zod + token validation cover this. |
| SSE `update` event payload → client store | The server's applyEvent result is pushed to the browser. The browser treats it as authoritative (mirrors `msg.state` into AppState). If an attacker could inject bogus update events, they could repaint the UI — but the SSE stream is cookie-authenticated (D-10) and Host-validated (D-11), so only local same-origin code can read/write it. |
| main.tsx URL parsing | `?token=...&session=...` comes from the server's launch URL. The token is captured and set via `setReviewToken(token)` BEFORE URL wipe; after wipe it lives only in memory. `sourceFromPrKey` parses the prKey via regex — regex failure falls through to local mode (documented limitation). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-2-04-01 | Tampering | Browser XSS via unescaped staleDiff content in the modal | mitigate | React auto-escapes text content in JSX. The modal only renders `storedSha.slice(0, 8)` and `currentSha.slice(0, 8)` — values that ultimately come from the server's `fetchCurrentHeadSha` result (git/gh output, SHA-shaped). Even a malformed SHA would be displayed as text, not executed. |
| T-2-04-02 | Tampering | CSRF on `/api/session/choose-resume` from a browser tab the user opened to a different site | mitigate | Server-side `tokenValidate` middleware enforces `X-Review-Token` header === cookie === launch token (double-submit). The token is captured into a module-level variable in api.ts BEFORE URL wipe; after wipe, there is no path for an external origin to read it (Same-Origin Policy prevents cross-origin reads of module state). |
| T-2-04-03 | Information Disclosure | Launch token lingers in `window.location` after page load | mitigate | `main.tsx` calls `history.replaceState('', '', '/')` AFTER `setReviewToken(token)`. The token is captured into module memory first; URL history no longer carries it. Document loads are not logged to DevTools history by default once the URL is replaced. |
| T-2-04-04 | Denial-of-Service | Modal renders unconditionally + consumes staleDiff from an update event — if server pushes invalid data, store crashes | mitigate | The store's onSnapshot/onUpdate path spreads `msg.state.staleDiff` directly (no parse). If server sends malformed data, TypeScript at compile time + structural Object spread prevent runtime throws. If `staleDiff` is undefined, the modal renders null (early return). No error path crashes the UI. |
| T-2-04-05 | Spoofing | Local-mode prKey cannot recover `source.base`/`source.head` in `sourceFromPrKey` | accept (documented limitation) | Phase 2 limitation: local prKeys are hash-based and cannot be reversed. "Discard session" still works (server doesn't need source for reset if it's just deleting state). "Refresh to current PR" on a local session may 400 because the server-side zod schema requires non-empty `base`/`head`. Documented in Plan 03 SUMMARY + in code comment. |
| T-2-04-06 | Denial-of-Service | User clicks modal button, network hangs → modal stuck in "Refreshing diff…" indefinitely | accept | No timeout in `chooseResume`. User can close the browser and restart. Adding a timeout is UX polish, not a security control. The AbortController pattern is a v2 concern. |
| T-2-04-07 | Information Disclosure | Error message in modal shows raw fetch error including status codes | accept | The modal displays `err.message` which contains things like "HTTP 500" — informative to the developer/user, no sensitive content (the server's error body is a generic "Resume failed" per Plan 03 mitigation T-2-03-05). |
