Simple deterministic paraphrase: take the first non-empty paragraph, truncate to ~280 chars, strip markdown noise (bullets, headers), collapse whitespace. If the description is empty or whitespace, return `(no description provided — review the changes below)`.

    Example `paraphrase` implementation:
    ```typescript
    function paraphrase(desc: string): string {
      if (!desc || !desc.trim()) return '(no description provided — review the changes below)';
      // Strip markdown: headers, bullets, HTML comments
      const stripped = desc
        .replace(/<!--[\s\S]*?-->/g, '')           // HTML comments
        .replace(/^#+\s+/gm, '')                    // Headers
        .replace(/^[-*+]\s+/gm, '')                 // Bullets
        .replace(/\[([^\]]+)\]\([^)]+\)/g, '$1')    // Markdown links → text
        .replace(/`{1,3}[^`]*`{1,3}/g, '')          // Inline code
        .replace(/\s+/g, ' ')
        .trim();
      const firstPara = stripped.split(/\n\n+/)[0] ?? stripped;
      return firstPara.length > 280 ? firstPara.slice(0, 277) + '...' : firstPara;
    }
    ```

    ```typescript
    import { describe, it, expect, afterAll } from 'vitest';
    import { execa, ExecaChildProcess } from 'execa';
    import { mkdtempSync, writeFileSync } from 'node:fs';
    import os from 'node:os';
    import path from 'node:path';

    describe('Phase 1 end-to-end', () => {
      let child: ExecaChildProcess | null = null;
      let repoDir = '';

      async function makeRepo() {
        const dir = mkdtempSync(path.join(os.tmpdir(), 'gitrev-e2e-'));
        await execa('git', ['init', '-q'], { cwd: dir });
        await execa('git', ['config', 'user.email', 't@t'], { cwd: dir });
        await execa('git', ['config', 'user.name', 't'], { cwd: dir });
        writeFileSync(path.join(dir, 'a.ts'), 'export const x = 1;\n');
        await execa('git', ['add', '.'], { cwd: dir });
        await execa('git', ['commit', '-qm', 'c1'], { cwd: dir });
        writeFileSync(path.join(dir, 'a.ts'), 'export const x = 2;\n');
        await execa('git', ['add', '.'], { cwd: dir });
        await execa('git', ['commit', '-qm', 'c2'], { cwd: dir });
        return dir;
      }

      afterAll(() => { if (child) child.kill('SIGTERM'); });

      it('boots, handles start_review via stdio, passes security probes, shuts down on SIGTERM', async () => {
        repoDir = await makeRepo();
        // Build is a prereq — test assumes `pnpm --filter server build` already ran
        child = execa('node', ['server/dist/index.js'], {
          cwd: repoDir,
          env: { ...process.env, CLAUDE_PLUGIN_DATA: path.join(repoDir, '.plugin-data') },
          reject: false, buffer: false,
        });

        // Read stderr until we see the listen URL (or 5s timeout)
        let stderrBuf = '';
        const listenUrlPromise = new Promise<string>((resolve, reject) => {
          const timeout = setTimeout(() => reject(new Error('timeout waiting for listen URL')), 5000);
          child!.stderr!.on('data', (chunk: Buffer) => {
            stderrBuf += chunk.toString();
            const m = stderrBuf.match(/http:\/\/127\.0\.0\.1:(\d+)/);
            if (m) { clearTimeout(timeout); resolve(m[0]); }
          });
        });
        const listenUrl = await listenUrlPromise;
        expect(listenUrl).toMatch(/^http:\/\/127\.0\.0\.1:\d+$/);
        const port = parseInt(listenUrl.match(/:(\d+)$/)![1]!, 10);

        // Send MCP initialize + tools/call via stdio
        const requests = [
          { jsonrpc: '2.0', id: 1, method: 'initialize', params: { protocolVersion: '2024-11-05', clientInfo: { name: 'e2e', version: '0' }, capabilities: {} } },
          { jsonrpc: '2.0', id: 2, method: 'tools/call', params: { name: 'start_review', arguments: { source: { kind: 'local', base: 'HEAD~1', head: 'HEAD' } } } },
        ];
        for (const r of requests) child!.stdin!.write(JSON.stringify(r) + '\n');

        // Collect stdout responses until we see id=2
        let stdoutBuf = '';
        const toolResultPromise = new Promise<object>((resolve, reject) => {
          const timeout = setTimeout(() => reject(new Error('timeout waiting for tools/call result. stderr: ' + stderrBuf.slice(-500))), 15000);
          child!.stdout!.on('data', (chunk: Buffer) => {
            stdoutBuf += chunk.toString();
            for (const line of stdoutBuf.split('\n')) {
              try {
                const parsed = JSON.parse(line);
                if (parsed.id === 2 && parsed.result) { clearTimeout(timeout); resolve(parsed.result); return; }
              } catch { /* partial line */ }
            }
          });
        });
        const result = await toolResultPromise as { content: Array<{ type: string; text: string }> };
        expect(result.content?.[0]?.text ?? '').toMatch(/Review open at: http:\/\/127\.0\.0\.1:\d+\/\?token=/);

        // Security probes against the live server
        const probe = await execa('bash', ['scripts/security-probes.sh', String(port)], { reject: false });
        expect(probe.exitCode).toBe(0);

        // SIGTERM shuts down cleanly
        child!.kill('SIGTERM');
        const exitCode = await new Promise<number>((resolve) => child!.on('exit', (c) => resolve(c ?? 1)));
        expect(exitCode).toBe(0);
      }, 30000);
    });
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Claude Code plugin loader → `.claude-plugin/plugin.json` | Trust is established here. Malformed or misplaced manifest means plugin silently fails to load. |
| `${CLAUDE_PLUGIN_ROOT}` env var → node args | Plugin loader substitutes before spawning. Hardcoding the absolute path would break when the plugin is installed elsewhere. |
| Slash command → MCP tool call | `commands/review.md` is a PROMPT (plain text). Claude Code expands it, the LLM then calls the MCP tool. The prompt itself is trusted content (authored by us); user input arrives via `$ARGUMENTS` substitution and is shape-checked by zod in the tool handler (Plan 02's discriminated union). |
| PR description text → LLM context | The paraphrased description enters the LLM's working memory via the tool call result. Potential prompt-injection surface if a malicious PR body contains instructions. Phase-1 mitigation: the description is TEXTUALLY PARAPHRASED (truncated + markdown-stripped), not LLM-interpreted, before being handed to Claude Code. The LLM sees it as data in a tool result, not as instructions. |
