    ```tsx
    // UI-SPEC §<DiffView>: thin wrapper around @git-diff-view/react. Unified mode only (D-23). Shiki tokens (D-22).
    // Pitfall 3/7: isolate the integration to this one file so a swap to react-diff-viewer-continued is local.
    import type { DiffModel, ShikiFileTokens } from '@shared/types';

    // Defensive import — the 0.1.3 package may expose DiffView as default OR named.
    // Task 1's spike identified which. If the spike showed `DiffView` as named export, this works.
    // If the spike showed `default`, swap the import line.
    import { DiffView as LibDiffView, DiffFile as LibDiffFile, DiffModeEnum } from '@git-diff-view/react';

    export function DiffView({ model, tokens }: { model: DiffModel; tokens: Record<string, ShikiFileTokens> }) {
      if (model.files.length === 0) return null;
      return (
        <div className="flex flex-col gap-[var(--spacing-xl)]">
          {model.files.map((file) => (
            <FileDiff key={file.id} file={file} fileTokens={tokens[file.id]} />
          ))}
        </div>
      );
    }

    function FileDiff({ file, fileTokens }: { file: DiffModel['files'][number]; fileTokens: ShikiFileTokens | undefined }) {
      // Reconstruct a unified-diff text chunk from the parsed hunks — the library accepts either
      // a diff string or a pre-parsed file object. String form is the most stable API surface across minor versions.
      const diffChunks = file.hunks.map((h) => {
        const lines = h.lines.map((l) => {
          const prefix = l.kind === 'add' ? '+' : l.kind === 'del' ? '-' : ' ';
          return `${prefix}${l.text}`;
        }).join('\n');
        return `${h.header}\n${lines}`;
      }).join('\n');

      // Minimal prop set — refine once the spike test output is consumed. If 0.1.3 requires a parsed
      // DiffFile instance, construct it via `new DiffFile(...)` here (the library commonly exposes
      // a DiffFile class for that purpose). If not present, fall back to the string path.
      try {
        // Try the class-based path first (library's documented modern shape)
        if (typeof LibDiffFile === 'function') {
          const df = new (LibDiffFile as unknown as new (path: string, diff: string) => unknown)(file.path, diffChunks);
          return (
            <div>
              <div className="font-mono" style={{ fontSize: 13, color: 'var(--color-text-secondary)', marginBottom: 'var(--spacing-sm)' }}>{file.path}</div>
              <LibDiffView data={df as any} mode={DiffModeEnum?.Unified ?? 1} />
            </div>
          );
        }
      } catch {
        // fall through to string-based render
      }

      return (
        <div>
          <div className="font-mono" style={{ fontSize: 13, color: 'var(--color-text-secondary)', marginBottom: 'var(--spacing-sm)' }}>{file.path}</div>
          {/* Fallback: render the raw unified-diff text inside a pre — at least the user sees SOMETHING */}
          <pre className="font-mono" style={{ fontSize: 13, color: 'var(--color-text)', backgroundColor: 'var(--color-surface-raised)', padding: 'var(--spacing-md)', borderRadius: 4, overflow: 'auto' }}>
            {diffChunks}
          </pre>
          {/* Shiki tokens reserved for the library-integrated path; the pre fallback doesn't consume them */}
          {fileTokens && <span data-testid="shiki-tokens-available" hidden />}
        </div>
      );
    }
    ```
